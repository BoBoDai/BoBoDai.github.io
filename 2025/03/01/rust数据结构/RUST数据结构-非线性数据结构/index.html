struct TreeNode<T> {
    data: T,
    children: Vec<TreeNode<T>>,
}

// 树：：节点指针
type Link<T> = Option<Box<T>>;
type Tree<T> = Link<TreeNode<T>>;

impl <T> TreeNode<T> {
    fn new(data: T) -> TreeNode<T> {
        TreeNode {
            data,
            children: Vec::new(),
        }
    }

    fn get_data(&self) -> &T {
        &self.data
    }

    fn get_child_nth(&self, i: usize) -> &TreeNode<T> {
        &self.children[i]
    }

    fn push_child(&mut self, child: TreeNode<T>) {
        self.children.push(child);
    }

    fn remove_child(&mut self, i: usize) -> TreeNode<T> {
        self.children.remove(i)
    }


}

二叉树

每个节点最多有两个子节点

实际上多叉树用长子兄弟表示法转换到二叉树

#[derive(Debug)]
struct BinaryTreeNode<T> {
    data: T,
    left: Link<BinaryTreeNode<T>>,
    right: Link<BinaryTreeNode<T>>,
}

// 树：：节点指针
type Link<T> = Option<Box<T>>;
type BinaryTree<T> = Link<BinaryTreeNode<T>>;

impl <T> BinaryTreeNode<T> {
    fn new(data: T) -> BinaryTreeNode<T> {
        BinaryTreeNode {
            data,
            left: None,
            right: None,
        }
    }

    fn get_data(&self) -> &T {
        &self.data
    }

    fn set_data(&mut self, data: T) {
        self.data = data;
    }

    fn get_left(&self) -> &BinaryTree<T> {
        &self.left
    }
    fn get_right(&self) -> &BinaryTree<T> {
        &self.right
    }

    fn set_lift(&mut self, child: BinaryTree<T>) {
        self.left = child;
    }

    fn set_right(&mut self, child: BinaryTree<T>) {
        self.right = child;
    }
}

计算二叉树的深度、节点数、叶子节点数

impl<T> BinaryTreeNode<T> {
    fn size(&self) -> usize {
        let mut size = 1;
        if let Some(left) = &self.left {
            size += left.size();
        }
        if let Some(right) = &self.right {
            size += right.size();
        }
        size
    }

    // 二叉树深度
    fn depth(&self) -> usize {
        let left_depth = if let Some(left) = &self.left {
            left.depth()
        } else {
            0
        };
        let right_depth = if let Some(right) = &self.right {
            right.depth()
        } else {
            0
        };
        1 + left_depth.max(right_depth)
    }

    // 计算叶子节点
    fn leaf_size(&self) -> usize {
        //左右都为空那么当前节点就是叶子节点
        if self.left.is_some() && self.right.is_some() {
            return 1;
        }
        let left_leaf = match &self.left {
            None => 0,
            Some(left) => left.leaf_size(),
        };
        let right_leaf = match &self.right {
            None => 0,
            Some(right) => right.leaf_size(),
        };
        left_leaf + right_leaf
    }

    // 计算非叶子节点数
    fn name_leaf_size(&self) -> usize {
        self.size() - self.leaf_size()
    }
}

二叉树的遍历

impl<T: std::fmt::Display> BinaryTreeNode<T> {
    //后序遍历
    fn pos_order(&self)
    where
        T: std::fmt::Display
    {
        self.left.as_ref().map(|n| n.pos_order());
        self.right.as_ref().map(|n| n.pos_order());
        println!("{}", self.data);
    }

    //前序遍历
    fn pre_order(&self)
    where
        T: std::fmt::Display
    {
        println!("{}", self.data);
        self.left.as_ref().map(|n| n.pre_order());
        self.right.as_ref().map(|n| n.pre_order());
    }

    //中序遍历
    fn in_order(&self)
    where
        T: std::fmt::Display
    {
        self.left.as_ref().map(|n| n.in_order());
        println!("{}", self.data);
        self.right.as_ref().map(|n| n.in_order());
    }
}

二叉搜索树

  • 在二叉树的计算上要求节点值有序：左节点的值 <= 当前节点 <= 右子树任意节点
  • 在维持平衡的情况下，具有O（logn）的随机查找、插入和删除性能。
  • 应用：实现 Ordered Map /Set 数据结构

二叉搜索树的查找

//查找
impl<T: Ord> BinarySearchTreeNode<T> {
    //查询data是否在树中
    fn contains(&self, data: &T) -> bool {
        use std::cmp::Ordering;

        match self.data.cmp(data) {
            Ordering::Less => match &self.right {
                None => false,
                Some(right) => right.contains(data),
            },
            Ordering::Equal => true,
            Ordering::Greater => match &self.left {
                None => false,
                Some(left) => left.contains(data),
            },
        }
    }
}

二叉搜索树的插入

//插入
impl<T: Ord> BinarySearchTreeNode<T> {
    fn insert(&mut self, data: T) {
        use std::cmp::Ordering;
        
        match &self.data.cmp(&data) {
            Ordering::Equal => {}
            Ordering::Greater => match &mut self.left {
                // 当前节点 > 目标 key，插入到左子树
                Some(left) => left.insert(data),
                None => self.set_lift(Some(Box::new(BinarySearchTreeNode::new(data))))
            },
            Ordering::Less => match &mut self.right {
                // 当前节点 < 目标 key，插入到右子树
                Some(right) => right.insert(data),
                None => self.set_right(Some(Box::new(BinarySearchTreeNode::new(data)))),
            }
        }
    }
}

二叉搜索树的删除

fn remove(mut self: Box<Self>, data: &T) -> Option<Box<BinarySearchTreeNode<T>>> {
    use std::cmp::Ordering;

    match &self.data.cmp(data) {
        Ordering::Equal => {
            //当前节点就是要删除的节点
            if self.right.is_none() {
                // 只有左子节点，将左子节点替换当前节点
                return self.left.take();
            }

            // 右子节点存在，找到右子节点中最小节点替换当前节点
            let mut leftmost = self.right.as_mut().unwrap();
            while let Some(ref mut next) = leftmost.left {
                leftmost = next;
            }
            std::mem::swap(&mut self.left, &mut leftmost.left);

            // 递归删除右子树中最小节点
            self.right = self.right.take().and_then(|right| right.remove(&data));
            Some(self)

        }
        Ordering::Greater => {
            //当前节点 > 目标 key， 在左子树查找
            if let Some(left) = self.left {
                self.left = left.remove(data);
            }
            Some(self)
        }
        Ordering::Less => {
            // 当前节点 < 目标key，在右子树中查找
            if let Some(right) = self.right {
                self.right = right.remove(data);
            }
            Some(self)
        }
    }
}

堆

一种特殊的树形数据结构
当前节点 <= 当前节点的所有孩子节点 ==> 最小堆
当前节点 >= 当前节点所有的孩子节点 ==> 最大堆

性质： O(1) 获取数组的最值

应用：实现优先级队列、排序算法（堆排序）、获取流式数据的最值

堆排序若一个节点处于下标p处，它的左节点处于2p处，右子节点处于2p + 1处。

// 二叉堆
#[derive(Debug, Clone)]
struct BinaryHeap<T: Ord> {
    data: Vec<T>,// 压缩储存
}

// 计算父节点的下标
macro_rules! parent {
    ($child:ident) => {$child >> 1};
}

// 计算左子节点的下标
macro_rules! left_child {
    ($parent:ident) => {$parent << 1};
}

//计算右子节点的下标
macro_rules! right_child {
    ($parent:ident) => {($parent << 1) + 1};
}

插入元素

impl<T: Ord> BinaryHeap<T> {
    fn new() -> Self {
        BinaryHeap {
            size: 0,
            data: vec![]
        }
    }

    // 获取堆顶元素
    fn top(&self) -> Option<&T> {
        self.data.get(0)
    }

    // 插入元素
    fn push(&mut self, val: T) {
        // 插入到末端
        self.data.push(val);

        let mut child = self.data.len() - 1;
        let mut parent = parent!(child);

        // 自上而下调整堆
        while child > 0 && self.data[parent] == self.data[child] {
            self.data.swap(parent, child);
            child = parent;
            parent = parent!(child);
        }
    }
}

邻接矩阵：二维表示顶点和顶点之间的关系，直观，易于理解，适合存储稠密图（边很多），空间复杂度 O（n^2）浪费空间、适合场景，顶点固定，频繁修改边
领接表：对于每个顶点只记录存在的边、空间复杂度O(n+e)、适合场景：顶点不固定、频繁遍历图、稀疏图
